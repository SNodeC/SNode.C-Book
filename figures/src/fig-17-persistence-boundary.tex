% ----------------------------------------------------------------------
% Transient communication side.
% ----------------------------------------------------------------------
\node[comm block] (peerConnection) {%
  \textbf{\textcolor{snodecTitleBlue}{peer connection}}\\[-0.4mm]
  socket lifecycle\\[-0.2mm]
  read/write events
};

\node[comm block, right=\colGap of peerConnection] (protocolContext) {%
  \textbf{\textcolor{snodecTitleBlue}{protocol context}}\\[-0.4mm]
  protocol meaning\\[-0.2mm]
  messages, requests
};

\node[app block, right=\colGap of protocolContext] (applicationState) {%
  \textbf{\textcolor{snodecTitleGreen}{application state}}\\[-0.4mm]
  what changed?\\[-0.2mm]
  what should survive?
};

\draw[flow] (peerConnection.east) -- (protocolContext.west);
\draw[flow] (protocolContext.east) -- (applicationState.west);

% ----------------------------------------------------------------------
% Durable state side.
% ----------------------------------------------------------------------
\node[persist block, below=\rowGap of peerConnection] (persistenceAdapter) {%
  \textbf{\textcolor{snodecTitleOrange}{persistence adapter}}\\[-0.4mm]
  mapping + policy\\[-0.2mm]
  schema knowledge
};

\node[persist block, below=\rowGap of protocolContext] (databaseClient) {%
  \textbf{\textcolor{snodecTitleOrange}{database client}}\\[-0.4mm]
  query execution\\[-0.2mm]
  result and error handling
};

\node[store block, below=\rowGap of applicationState] (durableStore) {%
  \textbf{\textcolor{snodecTitleNeutral}{database / store}}\\[-0.4mm]
  tables, indexes\\[-0.2mm]
  transactions
};

\draw[flow] (persistenceAdapter.east) -- (databaseClient.west);
\draw[flow] (databaseClient.east) -- (durableStore.west);

% Fitted lane panels.

  ] (transientPanel) {};

  \node[
    lane panel,
    fit=(persistenceAdapter)(databaseClient)(durableStore),

  ] (durablePanel) {};
\end{scope}

\node[panel label]
  at (transientPanel.north)
  {Transient communication side};

\node[panel label]

\coordinate (appBoundaryRoute)
  at (applicationState.south |- boundaryMidY);

\coordinate (adapterBoundaryRoute)
  at (persistenceAdapter.north |- boundaryMidY);

  at ($(appBoundaryRoute)!0.5!(adapterBoundaryRoute)$);

\draw[boundary line] (appBoundaryRoute) -- (adapterBoundaryRoute);
\draw[boundary flow] (adapterBoundaryRoute) -- (persistenceAdapter.north);

  at (boundaryLabelPos)
  {persistence boundary};
