  },

    draw=snodecOrange!75,

% ----------------------------------------------------------------------
% Configured server/client role.
% ----------------------------------------------------------------------
\node[role block] (role) {%
  \textbf{\textcolor{snodecGreen}{Server/client instance}}\\[-0.4mm]
  {\footnotesize configured communication role}\\[-0.2mm]
  {\footnotesize same application-side role: listen/connect, instance name, sections}
};

% ----------------------------------------------------------------------
% Endpoint family.
% ----------------------------------------------------------------------
\node[family block, below=\majorVGap of role] (family) {%
  \textbf{\textcolor{snodecBlue}{Endpoint family}}\\[-0.4mm]
  {\footnotesize address and carrier family stay the same}\\[-0.2mm]
  {\footnotesize \texttt{net::in}, \texttt{net::in6}, \texttt{net::un}, \texttt{net::rc}, \texttt{net::l2}}
};

% ----------------------------------------------------------------------
% Stream connection form.
% The larger placement gap compensates for the stream panel's fit inner sep,
% so the visible box-to-box gap stays consistent with the other layers.
% ----------------------------------------------------------------------
\node[stream title, below=\streamPlacementGap of family] (streamTitle) {%
  \textbf{\textcolor{snodecOrange}{Stream connection form}}
};

\node[variant block, below=\minorVGap of streamTitle, xshift=-\variantShift] (legacyStream) {%
  \textbf{\textcolor{snodecOrange!90!black}{Legacy stream}}\\[-0.4mm]
  plain byte transport\\[-0.2mm]
  no TLS context
};

\node[variant block, below=\minorVGap of streamTitle, xshift=\variantShift] (tlsStream) {%
  \textbf{\textcolor{snodecOrange!90!black}{TLS stream}}\\[-0.4mm]
  encrypted byte transport\\[-0.2mm]
  \texttt{ConfigTlsClient} / \texttt{ConfigTlsServer}
};

\coordinate (variantBottomCenter) at ($(legacyStream.south)!0.5!(tlsStream.south)$);

\node[stream note, below=\minorVGap of variantBottomCenter] (specializationNote) {%
  specialization point: connection setup, TLS context, certificates, SNI
};

\begin{scope}[on background layer]
  \node[
    stream panel,
    fit=(streamTitle)(legacyStream)(tlsStream)(specializationNote),
    inner sep=3mm
  ] (stream) {};
\end{scope}

% ----------------------------------------------------------------------
% Socket context.
% ----------------------------------------------------------------------
\node[context block, below=\majorVGap of stream] (context) {%
  \textbf{\textcolor{snodecRule}{Socket context}}\\[-0.4mm]
  {\footnotesize protocol behavior attached to the established stream connection}\\[-0.2mm]
  {\footnotesize HTTP, WebSocket, MQTT, custom protocol contexts}
};
