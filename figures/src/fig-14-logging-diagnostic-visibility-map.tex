% ----------------------------------------------------------------------
% Responsibility / visibility hierarchy.
% ----------------------------------------------------------------------
\node[app block] (application) {%
  \textbf{\textcolor{snodecBlue}{Application shell}}\\[-0.4mm]
  {\footnotesize operational program boundary}\\[-0.2mm]
  {\footnotesize config file, generated command line, logging, daemonization, quiet mode}\\[-0.2mm]
  {\footnotesize diagnostic evidence: effective configuration, output mode, startup state}
};

\node[instance block, below=\majorVGap of application] (instance) {%
  \textbf{\textcolor{snodecGreen}{Instance scope}}\\[-0.4mm]
  {\footnotesize configured server/client role visible at runtime}\\[-0.2mm]
  {\footnotesize enabled/disabled state, listen/connect activation, retry scheduling}\\[-0.2mm]
  {\footnotesize diagnostic evidence: role lifecycle, endpoint selection, activation result}
};

\node[connection block, below=\majorVGap of instance] (connection) {%
  \textbf{\textcolor{snodecOrange}{Connection scope}}\\[-0.4mm]
  {\footnotesize one concrete peer relationship}\\[-0.2mm]
  {\footnotesize connection name, local/remote addresses, queued/sent/read/processed data}\\[-0.2mm]
  {\footnotesize diagnostic evidence: peer episode, counters, duration, disconnect summary}
};

\node[context block, below=\majorVGap of connection] (context) {%
  \textbf{\textcolor{snodecRule}{Context scope}}\\[-0.4mm]
  {\footnotesize protocol behavior attached to one connection}\\[-0.2mm]
  {\footnotesize protocol state, accepted/rejected frames, application-level events}\\[-0.2mm]
  {\footnotesize diagnostic evidence: protocol meaning, state transitions, close decisions}
};

};

\node[form block, below=\minorVGap of formsLabel] (plogForm) {%
  \textbf{\texttt{PLOG(level)}}\\[-0.4mm]

};

\node[form block, below=\minorVGap of formsLabel, xshift=\formShift] (vlogForm) {%
