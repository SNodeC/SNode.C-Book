\begin{tikzpicture}[
  snodec diagram,
  semantics block/.style={
    draw=snodecBlue!75,
    fill=snodecBlueSoft,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=118mm,

    font=\sffamily\small
  },
  native block/.style={
    draw=snodecGreen!70,
    fill=snodecGreenSoft,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=54mm,
    inner xsep=3mm,
    inner ysep=2.4mm,
    font=\sffamily\small
  },
  websocket block/.style={
    draw=snodecOrange!75,
    fill=snodecOrangeSoft,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=54mm,
    inner xsep=3mm,
    inner ysep=2.4mm,
    font=\sffamily\small
  },
  native basis block/.style={
    draw=snodecGreen!55,
    fill=snodecGreenSoft!65,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=54mm,
    inner xsep=3mm,
    inner ysep=2.4mm,
    font=\sffamily\small
  },
  websocket basis block/.style={
    draw=snodecOrange!55,
    fill=snodecOrangeSoft!65,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=54mm,
    inner xsep=3mm,
    inner ysep=2.4mm,
    font=\sffamily\small
  },
  semantics flow/.style={
    snodec arrow,
    draw=snodecBlue!65
  },
  native flow/.style={
    snodec arrow,
    draw=snodecGreen!70
  },
  websocket flow/.style={
    snodec arrow,
    draw=snodecOrange!75
  },
  link/.style={
    draw=snodecMuted!75,
    line width=0.75pt
  },
  arrow label/.style={
    font=\sffamily\footnotesize,

  }
]

\node[semantics block] (semantics) {%
  \textbf{\textcolor{snodecBlue}{MQTT protocol semantics}}\\[-0.4mm]
  {\footnotesize CONNECT, SUBSCRIBE, PUBLISH, QoS, session state, topic flow}
};

\node[native block, below=\splitGap of semantics, xshift=-\carrierHalfSep] (nativeCarrier) {%
  \textbf{\textcolor{snodecGreen}{Native MQTT carrier}}\\[-0.4mm]
  {\footnotesize MQTT packets directly on stream}
};

\node[websocket block, below=\splitGap of semantics, xshift=\carrierHalfSep] (wsCarrier) {%
  \textbf{\textcolor{snodecOrange}{MQTT over WebSocket}}\\[-0.4mm]
  {\footnotesize MQTT packets inside WebSocket frames}
};

\coordinate (carrierBranch) at ([yshift=\branchGap]$(nativeCarrier.north)!0.5!(wsCarrier.north)$);
\coordinate (nativeBranch)  at ([yshift=\branchGap]nativeCarrier.north);
\coordinate (wsBranch)      at ([yshift=\branchGap]wsCarrier.north);

\draw[link]
  (semantics.south)
  -- node[arrow label, pos=0.38] {same MQTT packets and semantics}
  (carrierBranch);

\draw[link] (nativeBranch) -- (wsBranch);

\draw[semantics flow] (nativeBranch) -- (nativeCarrier.north);
\draw[semantics flow] (wsBranch) -- (wsCarrier.north);

\node[native basis block, below=\majorVGap of nativeCarrier] (streamBase) {%
  \textbf{\textcolor{snodecGreen!70!black}{Stream / TLS stream}}\\[-0.4mm]
  {\footnotesize lower family + transport}
};

\node[websocket basis block, below=\majorVGap of wsCarrier] (httpUpgrade) {%
  \textbf{\textcolor{snodecOrange!90!black}{HTTP upgrade path}}\\[-0.4mm]
  {\footnotesize HTTP handshake then WebSocket frames}
};

\draw[native flow] (nativeCarrier.south) -- (streamBase.north);
\draw[websocket flow] (wsCarrier.south) -- (httpUpgrade.north);

\node[native basis block, below=\majorVGap of streamBase] (endpoints) {%
  \textbf{\textcolor{snodecGreen!70!black}{Endpoint families}}\\[-0.4mm]
  {\footnotesize IPv4, IPv6, Unix, Bluetooth, \dots}
};

\node[websocket basis block, below=\majorVGap of httpUpgrade] (webCarrier) {%
  \textbf{\textcolor{snodecOrange!90!black}{Web-capable carrier}}\\[-0.4mm]
  {\footnotesize HTTP runs over stream / TLS stream}
};

\draw[native flow] (streamBase.south) -- (endpoints.north);
\draw[websocket flow] (httpUpgrade.south) -- (webCarrier.north);
