\documentclass[tikz,border=0pt]{standalone}
\input{snodec-figure-style.tex}
\usetikzlibrary{positioning,calc,arrows.meta,fit,backgrounds}

\begin{document}

\def\majorVGap{11mm}
\def\sideShift{39mm}

\begin{tikzpicture}[
  snodec diagram,
  boundary block/.style={
    draw=snodecBlue!65,
    fill=snodecBlueSoft!70,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=54mm,
    inner xsep=3mm,
    inner ysep=2.4mm,
    font=\sffamily\small
  },
  process block/.style={
    draw=snodecOrange!75,
    fill=snodecOrangeSoft,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=112mm,
    inner xsep=3mm,
    inner ysep=2.4mm,
    font=\sffamily\small
  },
  boundary flow/.style={
    snodec muted arrow,
    draw=snodecMuted!75,
    line width=0.75pt
  },
  arrow label/.style={
    font=\sffamily\footnotesize,
    text=snodecMuted,
    fill=white,
    inner sep=1.1pt
  }
]

% ----------------------------------------------------------------------
% Central running application process
% ----------------------------------------------------------------------
\node[process block] (process) {%
  \textbf{\textcolor{snodecOrange}{Running application process}}\\[-0.4mm]
  {\footnotesize event loop, registered instances, connections, contexts, state}
};

% ----------------------------------------------------------------------
% Top boundary roles
% ----------------------------------------------------------------------
\node[boundary block, above=\majorVGap of process, xshift=-\sideShift] (protocols) {%
  \textbf{\textcolor{snodecBlue}{Protocol-facing boundaries}}\\[-0.4mm]
  {\footnotesize servers, clients, bridges}\\[-0.2mm]
  {\footnotesize HTTP, MQTT, WebSocket, etc.}
};

\node[boundary block, above=\majorVGap of process, xshift=\sideShift] (configuration) {%
  \textbf{\textcolor{snodecBlue}{Configuration surface}}\\[-0.4mm]
  {\footnotesize CLI options, instance names}\\[-0.2mm]
  {\footnotesize files, defaults, profiles}
};

% ----------------------------------------------------------------------
% Bottom system concerns
% ----------------------------------------------------------------------
\node[boundary block, below=\majorVGap of process, xshift=-\sideShift] (visibility) {%
  \textbf{\textcolor{snodecBlue}{Operational visibility}}\\[-0.4mm]
  {\footnotesize logging, diagnostics}\\[-0.2mm]
  {\footnotesize counters, lifecycle}
};

\node[boundary block, below=\majorVGap of process, xshift=\sideShift] (deployment) {%
  \textbf{\textcolor{snodecBlue}{Deployment identity}}\\[-0.4mm]
  {\footnotesize executable, service}\\[-0.2mm]
  {\footnotesize package, host}
};

% ----------------------------------------------------------------------
% Relationships
% ----------------------------------------------------------------------
\draw[boundary flow]
  (protocols.south)
  -- node[arrow label, left=1.3mm] {communicates}
  (process.north -| protocols.center);

\draw[boundary flow]
  (configuration.south)
  -- node[arrow label, right=1.3mm] {configures}
  (process.north -| configuration.center);

\draw[boundary flow]
  (visibility.north)
  -- node[arrow label, left=1.3mm] {observes}
  (process.south -| visibility.center);

\draw[boundary flow]
  (deployment.north)
  -- node[arrow label, right=1.3mm] {runs as}
  (process.south -| deployment.center);


% ----------------------------------------------------------------------
% Fixed 160 mm output canvas.
% The drawing stays at native size and is centered horizontally.
% Vertical height remains natural, with the former standalone 7pt margin
% preserved above and below.
% ----------------------------------------------------------------------
\coordinate (snodecCanvasCenter) at (current bounding box.center);
\coordinate (snodecCanvasSW)
  at ($(snodecCanvasCenter)+(-80mm,0)$ |- current bounding box.south);
\coordinate (snodecCanvasNE)
  at ($(snodecCanvasCenter)+(80mm,0)$ |- current bounding box.north);
\path[use as bounding box]
  ([yshift=-7pt]snodecCanvasSW)
  rectangle
  ([yshift=7pt]snodecCanvasNE);

\end{tikzpicture}
\end{document}
