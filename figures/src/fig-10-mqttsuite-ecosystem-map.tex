  },
  broker block/.style={

    fill=snodecBlueSoft,

  },
  support block/.style={

  },

% ----------------------------------------------------------------------
% Center role
% ----------------------------------------------------------------------
\node[broker block] (broker) {%
  \textbf{\textcolor{snodecBlue}{\texttt{MQTTBroker}}}\\[-0.4mm]
  {\footnotesize topic space, subscriptions, retained messages, client state}
};

% ----------------------------------------------------------------------
% Top row: clients/tools and persistence-facing store
% ----------------------------------------------------------------------
\node[client block, above=\majorVGap of broker, xshift=-\sideShift] (clients) {%
  \textbf{\textcolor{snodecGreen}{MQTT clients and tools}}\\[-0.4mm]
  {\footnotesize native MQTT clients, devices,}\\[-0.2mm]
  {\footnotesize tools, services, browsers,}\\[-0.2mm]
  {\footnotesize dashboards, \texttt{MQTTCli}}
};

\node[store block, above=\majorVGap of broker, xshift=\sideShift] (store) {%
  \textbf{\textcolor{snodecOrange}{\texttt{MQTTStore}}}\\[-0.4mm]
  {\footnotesize persistent topic/state view,}\\[-0.2mm]
  {\footnotesize retained/state history,}\\[-0.2mm]
  {\footnotesize storage-facing role}
};

% ----------------------------------------------------------------------
% Bottom row: bridge and integration roles
% ----------------------------------------------------------------------
\node[support block, below=\majorVGap of broker, xshift=-\sideShift] (bridge) {%
  \textbf{\textcolor{snodecBlue}{\texttt{MQTTBridge}}}\\[-0.4mm]
  {\footnotesize broker-to-broker topic flow,}\\[-0.2mm]
  {\footnotesize prefixes and loop boundaries}
};

\node[support block, below=\majorVGap of broker, xshift=\sideShift] (integrator) {%
  \textbf{\textcolor{snodecBlue}{\texttt{MQTTIntegrator}}}\\[-0.4mm]
  {\footnotesize external services, databases,}\\[-0.2mm]
  {\footnotesize application-specific systems}
};

% ----------------------------------------------------------------------
% Interactions with the broker
% ----------------------------------------------------------------------
\draw[interaction flow]

  (broker.north -| clients.center);

\draw[interaction flow]
  (store.south)
  -- node[arrow label, right=1.3mm] {store / retrieve}
  (broker.north -| store.center);

\draw[interaction flow]
  (bridge.north)
  -- node[arrow label, left=1.3mm] {bridge / replicate}
  (broker.south -| bridge.center);

\draw[interaction flow]
  (integrator.north)
  -- node[arrow label, right=1.3mm] {integrate / adapt}
  (broker.south -| integrator.center);
