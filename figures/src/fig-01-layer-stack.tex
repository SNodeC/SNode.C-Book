\node[layer block] (family) {\textbf{\textcolor{snodecBlue}{Lower communication family}}\\[-0.5mm]
  {\footnotesize Endpoint family: IPv4, IPv6, Unix domain sockets, Bluetooth RFCOMM, or L2CAP.}
};

\node[layer block, above=6mm of family] (transport) {\textbf{\textcolor{snodecBlue}{Transport form}}\\[-0.5mm]

};

\node[layer block, above=6mm of transport] (connection) {\textbf{\textcolor{snodecBlue}{Connection handling}}\\[-0.5mm]

};

\node[layer block, above=6mm of connection] (protocol) {\textbf{\textcolor{snodecBlue}{Application protocol}}\\[-0.5mm]
  {\footnotesize Stream meaning: custom protocol, HTTP, WebSocket, MQTT, or another protocol layer.}
};

\node[layer block, above=6mm of protocol] (role) {\textbf{\textcolor{snodecBlue}{Application role}}\\[-0.5mm]
  {\footnotesize Endpoint job: server, client, bridge, broker, integrator, gateway, or custom role.}
};

\draw[stack arrow] (family.north) -- (transport.south);

  node[side note, midway, rotate=90, above=1.5mm] {more communication-specific};

  \node[fit=(family)(transport)(connection)(protocol)(role), snodec boundary, inner sep=5mm] (stack) {};
