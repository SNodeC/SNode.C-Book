\def\majorVGap{11mm}
\def\sideShift{39mm}

\begin{tikzpicture}[
  snodec diagram,
  boundary block/.style={
    draw=snodecBlue!65,
    fill=snodecBlueSoft!70,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=54mm,
    inner xsep=3mm,
    inner ysep=2.4mm,
    font=\sffamily\small
  },
  central block/.style={
    draw=snodecOrange!75,
    fill=snodecOrangeSoft,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=112mm,
    inner xsep=3mm,
    inner ysep=2.4mm,
    font=\sffamily\small
  },
  boundary flow/.style={
    snodec muted arrow,
    draw=snodecMuted!75,
    line width=0.75pt
  },
  arrow label/.style={
    font=\sffamily\footnotesize,
    text=snodecMuted,
    fill=white,
    inner sep=1.1pt
  }
]

% ----------------------------------------------------------------------
% Central application / gateway process
% ----------------------------------------------------------------------
\node[central block] (gateway) {%
  \textbf{\textcolor{snodecOrange}{IoT application / gateway process}}\\[-0.4mm]
  {\footnotesize shared state, policy, routing, translation, diagnostics}
};

% ----------------------------------------------------------------------
% Upper boundary actors
% ----------------------------------------------------------------------
\node[boundary block, above=\majorVGap of gateway, xshift=-\sideShift] (devices) {%
  \textbf{\textcolor{snodecBlue}{Field devices}}\\[-0.4mm]
  {\footnotesize sensors, actuators, nodes}\\[-0.2mm]
  {\footnotesize MQTT, Bluetooth, local links}
};

\node[boundary block, above=\majorVGap of gateway, xshift=\sideShift] (operator) {%
  \textbf{\textcolor{snodecBlue}{Browser / operator surface}}\\[-0.4mm]
  {\footnotesize pages, dashboards, controls}\\[-0.2mm]
  {\footnotesize HTTP, SSE, WebSocket}
};

% ----------------------------------------------------------------------
% Lower boundary resources
% ----------------------------------------------------------------------
\node[boundary block, below=\majorVGap of gateway, xshift=-\sideShift] (state) {%
  \textbf{\textcolor{snodecBlue}{Persistent state}}\\[-0.4mm]
  {\footnotesize database, measurements, history}\\[-0.2mm]
  {\footnotesize retained view}
};

\node[boundary block, below=\majorVGap of gateway, xshift=\sideShift] (services) {%
  \textbf{\textcolor{snodecBlue}{External services / brokers}}\\[-0.4mm]
  {\footnotesize MQTT broker, APIs, cloud}\\[-0.2mm]
  {\footnotesize other systems}
};

% ----------------------------------------------------------------------
% Boundary interactions
% ----------------------------------------------------------------------
\draw[boundary flow]
  (devices.south)
  -- node[arrow label, right=1.5mm] {telemetry / commands}
  (gateway.north -| devices.center);

\draw[boundary flow]
  (operator.south)
  -- node[arrow label, right=1.5mm] {observe / control}
  (gateway.north -| operator.center);

\draw[boundary flow]
  (state.north)
  -- node[arrow label, right=1.5mm] {state read/write}
  (gateway.south -| state.center);

\draw[boundary flow]
  (services.north)
  -- node[arrow label, right=1.5mm] {publish / integrate}
  (gateway.south -| services.center);

\begin{scope}[on background layer]
  \node[
    fit=(devices)(operator)(gateway)(state)(services),
    snodec boundary,
    inner sep=5mm,
    label={[font=\sffamily\bfseries\small,text=snodecBlue,fill=white,inner sep=1.2pt]above:IoT boundary constellation}
  ] {};
