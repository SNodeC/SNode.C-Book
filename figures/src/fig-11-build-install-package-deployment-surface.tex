\begin{document}

\def\majorVGap{11mm}
\def\hGap{8mm}

\begin{tikzpicture}[
  snodec diagram,
  build block/.style={
    draw=snodecBlue!75,
    fill=snodecBlueSoft,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=40mm,
    inner xsep=3mm,
    inner ysep=2.4mm,
    font=\sffamily\small
  },
  install block/.style={
    draw=snodecGreen!70,
    fill=snodecGreenSoft,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=40mm,
    inner xsep=3mm,
    inner ysep=2.4mm,
    font=\sffamily\small
  },
  package block/.style={
    draw=snodecOrange!75,
    fill=snodecOrangeSoft,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=40mm,
    inner xsep=3mm,
    inner ysep=2.4mm,
    font=\sffamily\small
  },
  surface block/.style={
    draw=snodecBlue!65,
    fill=snodecBlueSoft!60,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=140mm,
    inner xsep=3mm,
    inner ysep=2.4mm,
    font=\sffamily\small
  },
  service block/.style={
    draw=snodecGreen!70,
    fill=snodecGreenSoft,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=92mm,
    inner xsep=3mm,
    inner ysep=2.4mm,
    font=\sffamily\small
  },
  build flow/.style={
    snodec arrow,
    draw=snodecBlue!65,
    line width=0.75pt
  },
  install flow/.style={
    snodec arrow,
    draw=snodecGreen!70,
    line width=0.75pt
  },
  package flow/.style={
    snodec arrow,
    draw=snodecOrange!75,
    line width=0.75pt
  },
  deployment flow/.style={
    snodec muted arrow,
    draw=snodecMuted!75,
    line width=0.75pt
  },
  arrow label/.style={
    font=\sffamily\footnotesize,
    text=snodecMuted,
    fill=white,
    inner sep=1.1pt
  }
]

% ----------------------------------------------------------------------
% Build / install / package path
% ----------------------------------------------------------------------
\node[build block] (cmake) {%
  \textbf{\textcolor{snodecBlue}{CMake component}}\\[-0.4mm]
  {\footnotesize target, options,}\\[-0.2mm]
  {\footnotesize link dependencies}
};

\node[install block, right=\hGap of cmake] (install) {%
  \textbf{\textcolor{snodecGreen}{Install component}}\\[-0.4mm]
  {\footnotesize libraries, apps,}\\[-0.2mm]
  {\footnotesize exported config}
};

\node[package block, right=\hGap of install] (package) {%
  \textbf{\textcolor{snodecOrange}{Package component}}\\[-0.4mm]
  {\footnotesize package metadata,}\\[-0.2mm]
  {\footnotesize dependencies}
};

\draw[build flow] (cmake.east) -- (install.west);
\draw[install flow] (install.east) -- (package.west);

% ----------------------------------------------------------------------
% Deployed runtime surface
% ----------------------------------------------------------------------
\node[
  surface block,
  below=\majorVGap of install
] (surface) {%
  \textbf{\textcolor{snodecBlue}{Deployed runtime surface}}\\[-0.4mm]
  {\footnotesize libraries and executables, runtime-loaded modules, package exports}\\[-0.2mm]
  {\footnotesize service definition, configuration files, logs, pid files, writable state}
};

\coordinate (packageBendA) at ($(package.south)+(0,-5.5mm)$);
\coordinate (packageBendB) at (packageBendA -| surface.north);

\draw[package flow]
  (package.south)
  -- (packageBendA)
  -- (packageBendB)
  -- (surface.north);

\node[arrow label, left=1.3mm] at (packageBendB) {installs into};

% ----------------------------------------------------------------------
% Running service role
% ----------------------------------------------------------------------
\node[
  service block,
  below=\majorVGap of surface
] (service) {%
  \textbf{\textcolor{snodecGreen}{Running service role}}\\[-0.4mm]
  {\footnotesize complete runtime environment exists}
};

\draw[deployment flow]
  (surface.south)
  -- node[arrow label, left=1.3mm] {runs as}
  (service.north);


% ----------------------------------------------------------------------
% Fixed 160 mm output canvas.
% The drawing stays at native size and is centered horizontally.
% Vertical height remains natural, with the former standalone 7pt margin
% preserved above and below.
% ----------------------------------------------------------------------
\coordinate (snodecCanvasCenter) at (current bounding box.center);
\coordinate (snodecCanvasSW)
  at ($(snodecCanvasCenter)+(-80mm,0)$ |- current bounding box.south);
\coordinate (snodecCanvasNE)
  at ($(snodecCanvasCenter)+(80mm,0)$ |- current bounding box.north);
\path[use as bounding box]
  ([yshift=-7pt]snodecCanvasSW)
  rectangle
  ([yshift=7pt]snodecCanvasNE);
