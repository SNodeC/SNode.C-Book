\documentclass[tikz,border=7pt]{standalone}
\input{snodec-figure-style.tex}
\usetikzlibrary{positioning,calc,arrows.meta,fit,backgrounds}

\begin{document}

\def\majorVGap{11mm}
\def\minorVGap{4mm}
\def\sectionPlacementGap{14mm}
\def\viewShift{58mm}
\def\sectionItemGap{3.5mm}

\begin{tikzpicture}[
  snodec diagram,
  scope block/.style={
    draw=snodecBlue!72,
    fill=snodecBlueSoft,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=140mm,
    inner xsep=3mm,
    inner ysep=2.4mm,
    font=\sffamily\small
  },
  section panel/.style={
    draw=snodecGreen!70,
    fill=snodecGreenSoft,
    rounded corners=2.5pt,
    line width=0.75pt,
    inner xsep=3mm,
    inner ysep=2.4mm
  },
  option block/.style={
    draw=snodecOrange!75,
    fill=snodecOrangeSoft,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=140mm,
    inner xsep=3mm,
    inner ysep=2.4mm,
    font=\sffamily\small
  },
  section text/.style={
    align=center,
    text width=140mm,
    font=\sffamily\small
  },
  section item/.style={
    draw=snodecBlue!45,
    fill=snodecBlueSoft!35,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=23mm,
    inner xsep=2mm,
    inner ysep=1.5mm,
    font=\sffamily\footnotesize
  },
  view block/.style={
    draw=snodecBlue!45,
    fill=snodecBlueSoft!35,
    rounded corners=2.5pt,
    line width=0.75pt,
    align=center,
    text width=50mm,
    inner xsep=3mm,
    inner ysep=2.4mm,
    font=\sffamily\footnotesize
  },
  hierarchy flow/.style={
    snodec muted arrow,
    draw=snodecMuted!75,
    line width=0.75pt
  },
  view label/.style={
    font=\sffamily\footnotesize,
    text=snodecMuted,
    align=center
  }
]

% ----------------------------------------------------------------------
% Main hierarchy.
% ----------------------------------------------------------------------
\node[scope block] (application) {%
  \textbf{\textcolor{snodecBlue}{Application scope}}\\[-0.4mm]
  {\footnotesize executable-level operational shell}\\[-0.2mm]
  {\footnotesize config file, logging, daemonization, help, generated command line}
};

\node[scope block, below=\majorVGap of application] (instance) {%
  \textbf{\textcolor{snodecBlue}{Instance scope}}\\[-0.4mm]
  {\footnotesize configured server/client communication role}\\[-0.2mm]
  {\footnotesize external identity, disabled/enabled state}\\[-0.2mm]
  {\footnotesize e.g. \texttt{echo}, \texttt{mqtt-uplink}, \texttt{measurement-input}}
};

% ----------------------------------------------------------------------
% Section scope with centered concrete section names.
% The larger placement gap compensates for the section panel's fit inner sep,
% so the visible gap from Instance scope to Section scope matches the others.
% ----------------------------------------------------------------------
\node[section text, below=\sectionPlacementGap of instance] (sectionTitle) {%
  \textbf{\textcolor{snodecGreen}{Section scope}}\\[-0.4mm]
  {\footnotesize responsibility boundary inside one instance}
};

\node[section item, below=\minorVGap of sectionTitle] (connection) {\texttt{connection}};
\node[section item, left=\sectionItemGap of connection] (remote) {\texttt{remote}};
\node[section item, left=\sectionItemGap of remote] (local) {\texttt{local}};
\node[section item, right=\sectionItemGap of connection] (socket) {\texttt{socket}};
\node[section item, right=\sectionItemGap of socket] (tls) {\texttt{tls}};

\begin{scope}[on background layer]
  \node[
    section panel,
    fit=(sectionTitle)(local)(remote)(connection)(socket)(tls),
    inner sep=3mm
  ] (section) {};
\end{scope}

% ----------------------------------------------------------------------
% Concrete option.
% ----------------------------------------------------------------------
\node[option block, below=\majorVGap of section] (option) {%
  \textbf{\textcolor{snodecOrange}{Option}}\\[-0.4mm]
  {\footnotesize one concrete value inside a section}\\[-0.2mm]
  {\footnotesize host, port, path, timeout, retry, backlog, certificate path}
};

% ----------------------------------------------------------------------
% Hierarchy arrows.
% ----------------------------------------------------------------------
\draw[hierarchy flow] (application.south) -- (instance.north);
\draw[hierarchy flow] (instance.south) -- (section.north);
\draw[hierarchy flow] (section.south) -- (option.north);

% ----------------------------------------------------------------------
% Same hierarchy, three views.
% ----------------------------------------------------------------------
\node[view label, below=8mm of option] (viewsLabel) {Same hierarchy, three views};

\node[view block, below=\minorVGap of viewsLabel, xshift=-\viewShift] (cppApi) {%
  \textbf{C++ API}\\[-0.4mm]
  \texttt{getConfig()->Local::setPort(8080)}
};

\node[view block, below=\minorVGap of viewsLabel] (commandLine) {%
  \textbf{Command line}\\[-0.4mm]
  \texttt{app instance local --port 8080}
};

\node[view block, below=\minorVGap of viewsLabel, xshift=\viewShift] (configFile) {%
  \textbf{Config file}\\[-0.4mm]
  \texttt{instance.local.port = 8080}
};

% ----------------------------------------------------------------------
% Outer boundary.
% ----------------------------------------------------------------------
\begin{scope}[on background layer]
  \node[
    fit=(application)(instance)(section)(option)(viewsLabel)(cppApi)(commandLine)(configFile),
    snodec boundary,
    inner sep=5mm,
    label={[font=\sffamily\bfseries\small,text=snodecBlue,fill=white,inner sep=1.2pt]above:Configuration hierarchy}
  ] {};
\end{scope}

\end{tikzpicture}
\end{document}
