  flow/.style={snodec main arrow, draw=snodecMuted!76},

  feedback flow/.style={snodec secondary arrow, dashed, draw=snodecMuted!68},
  muted label/.style={snodec arrow label}

\node[core block, draw=snodecGreen!78, fill=snodecGreenSoft] (app) {%
  \makebox[\linewidth][c]{\textbf{\textcolor{snodecTitleGreen}{Application code}}}\\[-0.5mm]
  {\footnotesize Calls \texttt{init(...)}, registers roles, then starts or ticks the runtime.}
};

% Intentionally 11mm so the dashed runtime border sits visually halfway
% between the first and second box: 5.5mm above core::SNodeC and 5.5mm below Application code.
\node[core block, below=\snodecGapLarge of app] (snodec) {%
  \makebox[\linewidth][c]{\textbf{\textcolor{snodecTitleBlue}{core::SNodeC}}}\\[-0.5mm]
  {\footnotesize Public runtime control surface.}
};

\node[core block, below=\snodecGapSmall of snodec] (loop) {%
  \makebox[\linewidth][c]{\textbf{\textcolor{snodecTitleBlue}{core::EventLoop}}}\\[-0.5mm]
  {\footnotesize Orchestrates one event-processing step after another.}
};

\node[core block, below=\snodecGapSmall of loop] (mux) {%
  \makebox[\linewidth][c]{\textbf{\textcolor{snodecTitleBlue}{core::EventMultiplexer}}}\\[-0.5mm]
  {\footnotesize Waits for readiness, timers, and queued work.}
};

\node[source block, below=\snodecGapMajor of mux, xshift=-43mm] (descriptor) {%
  \textbf{\textcolor{snodecTitleGray}{Descriptors}}\\[-0.5mm]
  {\footnotesize socket and file-descriptor readiness}
};

\node[source block, below=\snodecGapMajor of mux] (timer) {%
  \textbf{\textcolor{snodecTitleGray}{Timers}}\\[-0.5mm]
  {\footnotesize scheduled and timeout activity}
};

\node[source block, below=\snodecGapMajor of mux, xshift=43mm] (queue) {%
  \textbf{\textcolor{snodecTitleGray}{Event queue}}\\[-0.5mm]
  {\footnotesize deferred runtime work}
};

\node[dispatch block, below=\snodecGapLarge of timer] (dispatch) {%
  \textbf{\textcolor{snodecTitleOrange}{Dispatch}}\\[-0.5mm]
  {\footnotesize Receivers, callbacks, and protocol contexts react.}
};

\draw[flow] (app.south) -- (snodec.north);
\draw[flow] (snodec.south) -- (loop.north);
\draw[flow] (loop.south) -- (mux.north);

\draw[source wait] (mux.south) -- ++(0,-\snodecBendLeg) -| (descriptor.north);
\draw[source wait] (mux.south) -- (timer.north);
\draw[source wait] (mux.south) -- ++(0,-\snodecBendLeg) -| (queue.north);

\draw[dispatch flow] (descriptor.south) |- (dispatch.west);
\draw[dispatch flow] (timer.south) -- (dispatch.north);
\draw[dispatch flow] (queue.south) |- ([yshift=2.5mm]dispatch.east);

\coordinate (feedbackLane) at ($(queue.east)+(\snodecOutsideLane,0)$);
\coordinate (feedbackBottom) at ($(dispatch.south)+(0,-\snodecBendLeg)$);
\draw[feedback flow]
  (dispatch.south)
  -- (feedbackBottom)
  -| (feedbackLane)
  |- (loop.east);

\begin{scope}[on background layer]
  \node[
    fit=(snodec)(loop)(mux)(descriptor)(timer)(queue)(dispatch)(feedbackLane)(feedbackBottom),
    snodec boundary,
    inner sep=\snodecBoundaryInnerSep
  ] (eventCoreBoundary) {};
\end{scope}
