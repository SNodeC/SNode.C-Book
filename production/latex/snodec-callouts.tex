% SNode.C book callout boxes.
% Restrained semantic boxes for publisher-style PDF output.
% No icons, no shadows, no decorative tabs: the title is integrated into the
% box body so repeated callouts do not dominate dense technical pages.

\usepackage[most]{tcolorbox}

\definecolor{snodecCalloutText}{HTML}{1F2933}

\definecolor{snodecRememberFrame}{HTML}{66819A}
\definecolor{snodecRememberBack}{HTML}{F5F9FC}
\definecolor{snodecNoteFrame}{HTML}{7D8790}
\definecolor{snodecNoteBack}{HTML}{F7F9FA}
\definecolor{snodecWarningFrame}{HTML}{A8793C}
\definecolor{snodecWarningBack}{HTML}{FFF8ED}
\definecolor{snodecRuleFrame}{HTML}{6F8B70}
\definecolor{snodecRuleBack}{HTML}{F4FAF4}
\definecolor{snodecChecklistFrame}{HTML}{8A9198}
\definecolor{snodecChecklistBack}{HTML}{F7F8F9}

\tcbset{
  snodecboxbase/.style={
    enhanced,
    breakable,
    arc=2.0pt,
    boxrule=0.45pt,
    left=6pt,
    right=6pt,
    top=5pt,
    bottom=5pt,
    before skip=9pt,
    after skip=9pt,
    fonttitle=\sffamily\bfseries\small,
    coltitle=snodecCalloutText,
    coltext=snodecCalloutText,
    detach title,
    before upper={\tcbtitle\par\vspace{2pt}\setlength{\parindent}{0pt}},
    parbox=false
  }
}

\newtcolorbox{snodecrememberbox}[1]{
  snodecboxbase,
  title={#1},
  colframe=snodecRememberFrame,
  colback=snodecRememberBack
}

\newtcolorbox{snodecnotebox}[1]{
  snodecboxbase,
  title={#1},
  colframe=snodecNoteFrame,
  colback=snodecNoteBack
}

\newtcolorbox{snodecwarningbox}[1]{
  snodecboxbase,
  title={#1},
  colframe=snodecWarningFrame,
  colback=snodecWarningBack
}

\newtcolorbox{snodecrulebox}[1]{
  snodecboxbase,
  title={#1},
  colframe=snodecRuleFrame,
  colback=snodecRuleBack
}

\newtcolorbox{snodecchecklistbox}[1]{
  snodecboxbase,
  title={#1},
  colframe=snodecChecklistFrame,
  colback=snodecChecklistBack
}
